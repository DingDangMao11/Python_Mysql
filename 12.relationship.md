```
#encoding: utf-8

from sqlalchemy import create_engine, Column, Integer, String, Table, ForeignKey
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker, relationship

HOSTNAME = '127.0.0.1'
PORT     = '3306'
DATABASE = 'first_sqlalchemy'
USERNAME = 'root'
PASSWORD = '0000'

DB_URI = "mysql+pymysql://{username}:{password}@{host}:{port}/{db}?charset=utf8".format\
    (username=USERNAME,password=PASSWORD,host=HOSTNAME,port=PORT,db=DATABASE)
# 创建数据库引擎
engine = create_engine(DB_URI)
Base = declarative_base(engine)
session = sessionmaker(engine)()

# 如何解决多对多关系？
# 用中间表连接两者
# 复合主键
# article
# id/title

# article_tag
#article_id/tag_id
# 1/1

# tag
# id/name
# 1/'tag1'
article_tag = Table(
    "article_tag",
    Base.metadata,
    Column("article_id",Integer,ForeignKey("article.id"),primary_key=True),
    Column("tag_id",Integer,ForeignKey("tag.id"),primary_key=True)
      )

class Article(Base):
    __tablename__ = 'article'
    id = Column(Integer, primary_key=True, autoincrement=True)
    title = Column(String(50), nullable=False)
     # 一篇文章下有多个Tag标签，backref：表示通过Tag拿到所有的文章，这个指引属性是articles，
    #  secondary：表示指向定义好的表Table的对象article_tag
    # 这样就将3张表关联起来
    # tags = relationship("Tag",backref="articles",secondary=article_tag)
    def __repr__(self):
        return "<Article(title:%s,content:%s)>" % (self.title,self.content)
class Tag(Base):
    __tablename__ = 'tag'
    id = Column(Integer, primary_key=True, autoincrement=True)
    name = Column(String(50), nullable=False)
    articles = relationship("Article",backref="tags",secondary=article_tag)
```
```
# 查询article文章下面的所有标签
article = session.query(Article).first()
print(article.tags)

# 得到标签下的所有文章
tag = session.query(Tag).first()
print(tag.articles)
```
